// include the library code:
#include <LiquidCrystal.h>

// initialize the library with the numbers of the interface pins
LiquidCrystal lcd(P2_0, P2_1, P2_2, P2_3, P2_4, P2_5);

#define boton_inc P1_3    //Boton incremento
#define boton_dec P1_4    //Boton decremento
#define boton_reset P1_5  //Boton reset

boolean estado_boton_inc;
boolean estado_boton_dec;
boolean estado_boton_reset;

boolean estado_boton_inc_ant;
boolean estado_boton_dec_ant;

byte contador=0;
float sensor;

char txt2stream[17];
String inputString;
boolean stringComplete;

void setup()
{
  pinMode(boton_inc, INPUT_PULLUP);
  pinMode(boton_dec,INPUT_PULLUP);
  pinMode(boton_reset,INPUT_PULLUP);
  pinMode(P1_6, OUTPUT);
  pinMode(P1_7, OUTPUT);
  pinMode(P2_7, OUTPUT);
  lcd.begin(16,2);
  delay(200);
  lcd.setCursor(0,0);
  lcd.print("I: D: R:  C:");
  lcd.setCursor(0,1);
  lcd.print("S:     L: M: B:");
  Serial.begin(9600);
}

void loop()
{
  estado_boton_inc = !digitalRead(boton_inc);
  estado_boton_dec = !digitalRead(boton_dec);
  estado_boton_reset = !digitalRead(boton_reset);

  if (estado_boton_inc!=estado_boton_inc_ant)
  {
     if (estado_boton_inc==1)
     {
        if (contador<100)
        {
           contador++;
        }
     }
     estado_boton_inc_ant = estado_boton_inc;
     delay(5);
  }

  sensor = analogRead(A0);
  sensor *= 100.0 / 1023.0;

  lcd.setCursor(2,0);
  sprintf(txt2stream,"%u",estado_boton_inc);
  lcd.print(txt2stream);

  lcd.setCursor(12,0);
  sprintf(txt2stream,"%u",contador);
  lcd.print(txt2stream);
  
  lcd.setCursor(2,1);
  lcd.print(sensor,1);

  Serial.print("$");
  Serial.print(estado_boton_inc);
  Serial.print("/");
  Serial.print(estado_boton_dec);
  Serial.print("/");
  Serial.print(estado_boton_reset);
  Serial.print("/");
  Serial.print(contador);
  Serial.print("/");
  Serial.println(sensor,1);

  if (stringComplete)
  {
    switch (inputString[0])
    {
      case '0': digitalWrite(P1_6, LOW); break;
      case '1': digitalWrite(P1_6, HIGH); break;
      default: break;
    }

    switch (inputString[1])
    {
      case '0': digitalWrite(P1_7, LOW); break;
      case '1': digitalWrite(P1_7, HIGH); break;
      default: break;
    }

    switch (inputString[2])
    {
      case '0': digitalWrite(P2_7, LOW); break;
      case '1': digitalWrite(P2_7, HIGH); break;
      default: break;
    }

    lcd.setCursor(9,1);
    sprintf(txt2stream,"%c",inputString[0]);
    lcd.print(txt2stream);

    // clear the string:
    inputString="";
    stringComplete = false;
  }
  delay(50);
}

void serialEvent()
{
  while (Serial.available()) {
    char inChar[2];
    // get the new byte:
    inChar[0] = Serial.read();
    // add it to the inputString:
    inputString += inChar[0];
    // if the incoming character is a newline, set a flag
    // so the main loop can do something about it:
    if (inChar[0] == '\r') {
    stringComplete = true;
    }
  }
}

        void SerialPort1DataReceived(object sender, System.IO.Ports.SerialDataReceivedEventArgs e)
        {
            int cant_datos = 5;
            string datos;
            string[] vector_datos;
            int pulsador_inc;
            int pulsador_dec;
            int pulsador_reset;
            int contador;
            double sensor;

            if (serialPort1.BytesToRead > 1)
            {
                datos = serialPort1.ReadLine();
                if (datos.StartsWith("$"))
                {
                    vector_datos = datos.Substring(1).Split('/');
                    if (vector_datos.Length==cant_datos)
                    {
                        pulsador_inc = Convert.ToInt16(vector_datos[0]);
                        pulsador_dec = Convert.ToInt16(vector_datos[1]);
                        pulsador_reset = Convert.ToInt16(vector_datos[2]);
                        contador = Convert.ToInt16(vector_datos[3]);
                        sensor = Convert.ToDouble(vector_datos[4])/10;
                        label2.Text = pulsador_inc.ToString();
                        label5.Text = pulsador_dec.ToString();
                        label8.Text = pulsador_reset.ToString();
                        label11.Text = contador.ToString();
                        label13.Text = sensor.ToString();
                        if (pulsador_inc==1)
                            label3.BackColor = Color.LimeGreen;
                        else
                            label3.BackColor = Color.DarkGray;
                        if (pulsador_dec==1)
                            label6.BackColor = Color.LimeGreen;
                        else
                            label6.BackColor = Color.DarkGray;
                        if (pulsador_reset==1)
                            label9.BackColor = Color.LimeGreen;
                        else
                            label9.BackColor = Color.DarkGray;
                        trackBar1.Value = contador;
                        progressBar1.Value = Convert.ToInt16(sensor);
                    }
                }
            }
        }
