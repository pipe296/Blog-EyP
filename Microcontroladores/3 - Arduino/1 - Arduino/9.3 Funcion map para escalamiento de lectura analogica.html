// include the library code:
#include <LiquidCrystal.h>

const int rs = 12, en = 11, d4 = 5, d5 = 4, d6 = 3, d7 = 2;
LiquidCrystal lcd(rs, en, d4, d5, d6, d7);

float sensor;

char txt2stream[10];

void setup()
{
  pinMode(10, OUTPUT);
  digitalWrite(10, HIGH);
  lcd.begin(16, 2);
  delay(200);
  lcd.setCursor(0,0);
  lcd.print("Sensor:");
}

void loop()
{
  sensor = analogRead(A0);
  sensor = map(sensor,0,1023,0,100);
  lcd.setCursor(8,0);
  lcd.print(sensor,1);
  lcd.print("   ");
  delay(500);
}
