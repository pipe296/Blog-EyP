// include the library code:
#include <LiquidCrystal.h>

const int rs = 12, en = 11, d4 = 5, d5 = 4, d6 = 3, d7 = 2;
LiquidCrystal lcd(rs, en, d4, d5, d6, d7);

#define boton_inc A1    //Boton incremento
#define boton_dec A2    //Boton decremento
#define boton_reset A3  //Boton reset

#define estado_boton_inc digitalRead(boton_inc)
#define estado_boton_dec digitalRead(boton_dec)
#define estado_boton_reset digitalRead(boton_reset)

boolean estado_boton_inc_ant;
boolean estado_boton_dec_ant;

byte contador = 0;
float sensor;

char inputString[10];
boolean stringComplete;
char strPWM[6];
int intPWM = 0;
float floatPWM = 0;

void setup()
{
  pinMode(A0, INPUT);
  pinMode(A1, INPUT);
  pinMode(A2, INPUT);
  pinMode(A4, OUTPUT);
  pinMode(A5, OUTPUT);
  pinMode(13, OUTPUT);
  digitalWrite(10, HIGH);
  lcd.begin(20,4);
  delay(200);
  lcd.setCursor(0,0);
  lcd.print("I:  D:  R:  C: ");
  lcd.setCursor(0,1);
  lcd.print("Sensor: ");
  lcd.setCursor(0,2);
  lcd.print("L:  M:  B: ");
  lcd.setCursor(0,3);
  lcd.print("PWM: ");
  Serial.begin(9600);
}

void loop() {
  if (estado_boton_inc!=estado_boton_inc_ant)
  {
    if (estado_boton_inc==1)
    {
      if (contador<100)
      {
        contador++;
      }
    }
    estado_boton_inc_ant = estado_boton_inc;
    delay(5);
  }

  //Programar acción boton_dec

  //Programar acción boton_reset

  sensor = analogRead(A0);
  sensor *= 100.0 / 1023.0;

  lcd.setCursor(2,0);
  lcd.print(estado_boton_inc);

  //Programar visualización estado_boton_dec

  //Programar visualización estado_boton_reset

  lcd.setCursor(14,0);
  lcd.print(contador);
  lcd.print("  ");

  lcd.setCursor(8,1);
  lcd.print(sensor,1);
  lcd.print("  ");

  Serial.print("$");
  Serial.print(estado_boton_inc);
  Serial.print("/");
  Serial.print(estado_boton_dec);
  Serial.print("/");
  Serial.print(estado_boton_reset);
  Serial.print("/");
  Serial.print(contador);
  Serial.print("/");
  Serial.println(sensor,1);

  if (stringComplete)
  {
    switch (inputString[0])
    {
      case '0': digitalWrite(A4, LOW); break;
      case '1': digitalWrite(A4, HIGH); break;
      default: break;
    }
    switch (inputString[1])
    {
      case '0': digitalWrite(A5, LOW); break;
      case '1': digitalWrite(A5, HIGH); break;
      default: break;
    }
      
    //Programar acción inputString[2]
   
    strPWM[0] = inputString[3];
    strPWM[1] = inputString[4];
    strPWM[2] = inputString[5];
    strPWM[3] = inputString[6];
    strPWM[4] = '\0';
    intPWM = atol(strPWM);
    floatPWM = intPWM / 10.0;
    analogWrite(9,intPWM);

    lcd.setCursor(2,2);
    lcd.print(inputString[0]);

    lcd.setCursor(6,2);
    lcd.print(inputString[1]);

    //Programar visualización inputString[2]

    lcd.setCursor(5,3);
    lcd.print(floatPWM,1);
    lcd.print("% ");

    // clear the string:
    sprintf(inputString,"");
    stringComplete = false;
  }
  delay(100);
}

void serialEvent() {
  while (Serial.available())
  {
    char inChar[2]="0";
    // get the new byte:
    inChar[0] = Serial.read();
    // add it to the inputString:
    strcat(inputString,inChar);
    // if the incoming character is a newline, set a flag
    // so the main loop can do something about it:
    if (inChar[0] == '\r')
    {
      stringComplete = true;
    }
  }
}
