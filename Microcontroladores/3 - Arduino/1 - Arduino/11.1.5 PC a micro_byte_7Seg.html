char inputString[10];
boolean stringComplete;

int pins_PB[7] = {8, 7, 6, 5, 4, 3, 2};

void output_PB(byte cifra) {
  byte i;
  for (i=0; i<7; i++)
  {
    digitalWrite(pins_PB[i], (cifra>>i)&1);
  }
}

void setup()
{
  byte i;
  for (i=0; i<7; i++)
  {
    pinMode(pins_PB[i], OUTPUT);
    digitalWrite(pins_PB[i], LOW);
  }
  Serial.begin(9600);
}

void loop() {
  if (stringComplete)
  {
    switch (inputString[0])
    {
      case '0': output_PB(0b00111111); break;
      case '1': output_PB(6); break;
      case '2': output_PB(91); break;
      case '3': output_PB(79); break;
      case '4': output_PB(102); break;
      case 53: output_PB(109); break;
      case 54: output_PB(125); break;
      case 55: output_PB(7); break;
      case 56: output_PB(127); break;
      case 57: output_PB(103); break;
      //Escribir código para procesar A-F
    }
    // clear the string:
    sprintf(inputString,"");
    stringComplete = false;
  }
}

void serialEvent() {
  while (Serial.available())
  {
    char inChar[2]="0";
    // get the new byte:
    inChar[0] = Serial.read();
    // add it to the inputString:
    strcat(inputString,inChar);
    // if the incoming character is a newline, set a flag
    // so the main loop can do something about it:
    if (inChar[0] == '\r')
    {
      stringComplete = true;
    }
  }
}
