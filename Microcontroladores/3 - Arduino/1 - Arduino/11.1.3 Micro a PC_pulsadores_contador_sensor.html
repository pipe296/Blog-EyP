// include the library code:
#include <LiquidCrystal.h>

const int rs = 12, en = 11, d4 = 5, d5 = 4, d6 = 3, d7 = 2;
LiquidCrystal lcd(rs, en, d4, d5, d6, d7);

#define boton_inc A1    //Boton incremento
#define boton_dec A2    //Boton decremento
#define boton_reset A3  //Boton reset

#define estado_boton_inc digitalRead(boton_inc)
#define estado_boton_dec digitalRead(boton_dec)
#define estado_boton_reset digitalRead(boton_reset)

boolean estado_boton_inc_ant;
boolean estado_boton_dec_ant;

byte contador = 0;
float sensor;

void setup()
{
  pinMode(boton_inc, INPUT);
  pinMode(boton_dec, INPUT);
  pinMode(boton_reset, INPUT);
  digitalWrite(10, HIGH);
  lcd.begin(16,2);
  delay(200);
  lcd.setCursor(0,0);
  lcd.print("BI:   BD:   BR:");
  lcd.setCursor(0,1);
  lcd.print("Con:    Sen:");
  Serial.begin(9600);
}

void loop()
{
  if (estado_boton_inc!=estado_boton_inc_ant)
  {
    if (estado_boton_inc==1)
    {
      if (contador<100)
      {
        contador++;
      }
    }
    estado_boton_inc_ant = estado_boton_inc;
    delay(5);
  }

  //Programar acción boton_dec

  //Programar acción boton_reset

  sensor = analogRead(A0);
  sensor *= 100.0 / 1023.0;

  lcd.setCursor(3,0);
  lcd.print(estado_boton_inc);

  //Programar visualización estado_boton_dec

  //Programar visualización estado_boton_reset

  lcd.setCursor(4,1);
  lcd.print(contador);
  lcd.print("  ");

  lcd.setCursor(12,1);
  lcd.print(sensor,1);
  lcd.print("  ");

  Serial.print("$");
  Serial.print(estado_boton_inc);
  Serial.print("/");
  Serial.print(estado_boton_dec);
  Serial.print("/");
  Serial.print(estado_boton_reset);
  Serial.print("/");
  Serial.print(contador);
  Serial.print("/");
  Serial.println(sensor,1);
  delay(100);
}

        void Button_BuscarClick(object sender, EventArgs e)
        {
            comboBox_Puertos.Items.Clear();
            foreach (string s in SerialPort.GetPortNames())
                comboBox_Puertos.Items.Add(s);
            if (SerialPort.GetPortNames().GetLength(0)>0)
                comboBox_Puertos.Enabled = true;
            serialPort1.Close();
        }

        void ComboBox_PuertosSelectedIndexChanged(object sender, EventArgs e)
        {
            if (comboBox_Puertos.Text != "")
            button_Conectar.Enabled = true;
        }
        
        void Button_ConectarClick(object sender, EventArgs e)
        {
            if (comboBox_Puertos.Text != "")
            {
                CheckForIllegalCrossThreadCalls = false;
                serialPort1.PortName = comboBox_Puertos.Text;
                comboBox_Puertos.Enabled = false;
                serialPort1.Open();
                button_Conectar.Enabled = false;
            }
        }
        
        void SerialPort1DataReceived(object sender, System.IO.Ports.SerialDataReceivedEventArgs e)
        {
            int cant_datos = 5;
            string datos;
            string[] vector_datos;
            int pulsador_inc;
            int pulsador_dec;
            int pulsador_reset;
            int contador;
            double sensor;

            if (serialPort1.BytesToRead > 1)
            {
                datos = serialPort1.ReadLine();
                if (datos.StartsWith("$"))
                {
                    vector_datos = datos.Substring(1).Split('/');
                    if (vector_datos.Length==cant_datos)
                    {
                        pulsador_inc = Convert.ToInt16(vector_datos[0]);
                        label2.Text = pulsador_inc.ToString();
                        if (pulsador_inc==1)
                            label3.BackColor = Color.LimeGreen;
                        else
                            label3.BackColor = Color.DarkGray;

                        pulsador_dec = Convert.ToInt16(vector_datos[1]);
                        label5.Text = pulsador_dec.ToString();
                        if (pulsador_dec==1)
                            label6.BackColor = Color.LimeGreen;
                        else
                            label6.BackColor = Color.DarkGray;

                        //Programar acción pulsador_reset

                        contador = Convert.ToInt16(vector_datos[3]);
                        label11.Text = contador.ToString();
                        trackBar1.Value = contador;

                        sensor = Convert.ToDouble(vector_datos[4])/10;
                        progressBar1.Value = Convert.ToInt16(sensor);
                        label13.Text = sensor.ToString();
                    }
                }
            }
        }
