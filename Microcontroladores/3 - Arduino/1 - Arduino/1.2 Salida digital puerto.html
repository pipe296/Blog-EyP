Salida digital puerto 

byte unidades = 0;

int pins_PD[8] = {0, 1, 2, 3, 4, 5, 6, 7};

void output_PD(byte cifra) {
  byte i;
  for (i=0; i<8; i++)
  {
    digitalWrite(pins_PD[i], (cifra>>i)&1);
  }
}

void setup() {
  // put your setup code here, to run once:
  byte i;
  for (i=0; i<8; i++)
  {
    pinMode(pins_PD[i], OUTPUT);
    digitalWrite(pins_PD[i], LOW);
  }
}

void loop() {
  // put your main code here, to run repeatedly:
  output_PD(unidades);
  unidades++;
  delay(1000);
}
