Entrada digital puerto

 

int pins_PA[6] = {A0, A1, A2, A3, A4, A5};
int pins_PB[6] = {8, 9, 10, 11, 12, 13};

byte input_PA() {
   byte cifra=0;
   byte i;
   for (i=0; i<6; i++)
   {
      cifra=cifra+(digitalRead(pins_PA[i])*(1<<i));
   }
   return cifra;
}

void output_PB(byte cifra) {
  byte i;
  for (i=0; i<6; i++)
  {
    digitalWrite(pins_PB[i], (cifra>>i)&1);
  }
}

void setup() {
  // put your setup code here, to run once:
  byte i;
  for (i=0; i<6; i++)
  {
    pinMode(pins_PA[i], INPUT);
    pinMode(pins_PB[i], OUTPUT);
    digitalWrite(pins_PB[i], LOW);
  }
}

void loop() {
  // put your main code here, to run repeatedly:
  output_PB(input_PA());
}
