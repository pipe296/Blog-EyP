Contador 0 - 99 multiplexado sin decodificador

 

 

#include <16F877A.h>

//#include <18F4550.h>

 

#fuses HS,NOWDT,NOPROTECT,NOLVP,NODEBUG

#use delay(clock=20000000)

 

#define boton_inc input(PIN_A0) //Definimos la ubicación de los pulsadores

#define boton_dec input(PIN_A1)

#define boton_reset input(PIN_A2)

 

int1 boton_inc_ant;

int1 boton_dec_ant;

signed int8 unidades=0;

signed int8 decenas=0;

 

void numero_a_segmentos(int8 numero)

{

   switch (numero)

   {

      case 0: OUTPUT_B(0b0111111); break;

      case 1: OUTPUT_B(0b0000110); break;

      case 2: OUTPUT_B(0b1011011); break;

      case 3: OUTPUT_B(0b1001111); break;

      case 4: OUTPUT_B(0b1100110); break;

      case 5: OUTPUT_B(0b1101101); break;

      case 6: OUTPUT_B(0b1111101); break;

      case 7: OUTPUT_B(0b0000111); break;

      case 8: OUTPUT_B(0b1111111); break;

      case 9: OUTPUT_B(0b1100111); break;

      default: OUTPUT_B(0b00000000); break;

   }

}

 

void main()

{

   while (true) //Bucle infinito que mantendrá el programa en ejecución

   {

      if (boton_inc_ant!=boton_inc) //Rutina de conteo ascendente

      {

         if (boton_inc==1) //Instrucción para evitar conteo

         {

            unidades++; //múltiple en una sola pulsación (antirrebote)

            if ((unidades>9)&&(decenas<9)) {unidades=0; decenas++;}

            if ((unidades>9)&&(decenas>=9)) {unidades=9; decenas=9;}

         }

         boton_inc_ant = boton_inc;

      }

 

      if (boton_dec_ant!=boton_dec) //Rutina de conteo descendente

      {

         if (boton_dec==1) //Instrucción para evitar conteo

         {

            unidades--; //múltiple en una sola pulsación (antirrebote)

            if ((unidades<0)&&(decenas>0)) {unidades=9; decenas--;}

            if ((unidades<0)&&(decenas<=0)) {unidades=0; decenas=0;}

         }

         boton_dec_ant = boton_dec;

      }

 

      if (boton_reset==1) //Rutina de reset

      {

         unidades=0;

         decenas=0;

      }

 

      numero_a_segmentos(unidades);

      OUTPUT_LOW(PIN_B7);

      delay_ms(50);

      numero_a_segmentos(decenas);

      OUTPUT_HIGH(PIN_B7);

      delay_ms(50);

   }

}
