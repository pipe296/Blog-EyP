#include <16F877A.h>

//#include <18F4550.h>

 

#fuses HS,NOWDT,NOPROTECT,NOLVP,NODEBUG

#use delay(clock=20000000)

 

void main()

{

while (true) //Bucle infinito que mantendrá el programa en ejecución

{

//Realiza la operación lógica OR entre las entradas en los pines A0 y A1

if ((INPUT(PIN_A0)==1)||(INPUT(PIN_A1)==1))

OUTPUT_HIGH(PIN_B0);

else

OUTPUT_LOW(PIN_B0);

}

}
