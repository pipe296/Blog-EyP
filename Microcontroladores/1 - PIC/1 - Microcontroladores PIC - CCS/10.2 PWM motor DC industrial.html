//#include <16F877A.h>

#include <16F877.h>

 

#fuses HS,NOWDT,NOPROTECT,NOLVP,NODEBUG

#use delay(clock=12000000)

 

#use PWM(OUTPUT=PIN_C2, FREQUENCY=10kHz, DUTY=0)

 

#include "./Drivers/flex_lcd.c" //Driver para LCD alfanumérico 2x16

 

int32 PWM;

 

void main()

{

   setup_adc_ports(AN0);           //Configura entrada analógica (sAN0 en el 16F887)

   setup_adc(adc_clock_internal);  //Establece frecuencia de muestreo conversor A/D

   set_adc_channel(0);             //Selecciona canal análogo a leer

   lcd_init();                     //Inicialización del LCD

   delay_ms(200);

   OUTPUT_HIGH(PIN_D3);            //El pin D3 es para la retroiluminación del LCD

   pwm_on();

   printf(lcd_putc,"PWM: 0");      //Se imprime el texto en el LCD

   while (true)                    //Bucle infinito que mantendrá el programa en ejecución

   {

      PWM=read_adc();

      delay_ms(10);

      PWM=PWM*1000/255;

      pwm_set_duty_percent(PWM);   //Establecer ciclo de trabajo PWM (0 - 1000)

      lcd_gotoxy(6,1);

      printf(lcd_putc,"%Ld ",PWM);

      delay_ms(500);

   }

}
