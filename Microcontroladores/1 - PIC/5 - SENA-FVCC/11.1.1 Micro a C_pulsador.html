#define boton_inc PIN_A1   //Boton incremento

#define boton_dec PIN_A2   //Boton decremento

#define boton_reset PIN_A3 //Boton reset

 

int1 estado_boton_inc;

int1 estado_boton_dec;

int1 estado_boton_reset;

 

char txt2stream[17];

 

void setup()

{

   pinMode(boton_inc, INPUT_);

   pinMode(PIN_D3, OUTPUT_);

   digitalWrite(PIN_D3, HIGH);

   lcd_begin(16,2);

   delay(200);

   lcd_setCursor(0,0);

   sprintf(txt2stream,"BI:   BD:   BR:");

   lcd_print(txt2stream);

   Serial_begin(9600);

}

 

void loop()

{

   estado_boton_inc = digitalRead(boton_inc);

 

   lcd_setCursor(3,0);

   sprintf(txt2stream,"%u",estado_boton_inc);

   lcd_print(txt2stream);

 

   sprintf(txt2stream,"$%u/%u/%u",estado_boton_inc,estado_boton_dec,estado_boton_reset);

   Serial_println(txt2stream);

   delay(50);

}

        void Button_BuscarClick(object sender, EventArgs e)
        {
            comboBox_Puertos.Items.Clear();
            foreach (string s in SerialPort.GetPortNames())
                comboBox_Puertos.Items.Add(s);
            if (SerialPort.GetPortNames().GetLength(0)>0)
                comboBox_Puertos.Enabled = true;
            serialPort1.Close();
        }

        void ComboBox_PuertosSelectedIndexChanged(object sender, EventArgs e)
        {
            if (comboBox_Puertos.Text != "")
            button_Conectar.Enabled = true;
        }
        
        void Button_ConectarClick(object sender, EventArgs e)
        {
            if (comboBox_Puertos.Text != "")
            {
                CheckForIllegalCrossThreadCalls = false;
                serialPort1.PortName = comboBox_Puertos.Text;
                comboBox_Puertos.Enabled = false;
                serialPort1.Open();
                button_Conectar.Enabled = false;
            }
        }
        
        void SerialPort1DataReceived(object sender, System.IO.Ports.SerialDataReceivedEventArgs e)
        {
            int cant_datos = 3;
            string datos;
            string[] vector_datos;
            int pulsador_inc;

            if (serialPort1.BytesToRead > 1)
            {
                datos = serialPort1.ReadLine();
                if (datos.StartsWith("$"))
                {
                    vector_datos = datos.Substring(1).Split('/');
                    if (vector_datos.Length==cant_datos)
                    {
                        pulsador_inc = Convert.ToInt16(vector_datos[0]);
                        label2.Text = pulsador_inc.ToString();
                        if (pulsador_inc==1)
                            label3.BackColor = Color.LimeGreen;
                        else
                            label3.BackColor = Color.DarkGray;
                    }
                }
            }
        }

        void MainFormLoad(object sender, EventArgs e)
        {
            System.Threading.Thread.CurrentThread.CurrentCulture = new System.Globalization.CultureInfo("en-US");
        }

        void MainFormFormClosed(object sender, FormClosedEventArgs e)
