#include <Arduino.h.encrypted>

 

void setup() {

   pinMode(PIN_A0, INPUT_);

   pinMode(PIN_B0, OUTPUT_);

}

 

void loop() {

   if (digitalRead(PIN_A0)==1)

      digitalWrite(PIN_B0, HIGH);

   else

      digitalWrite(PIN_B0, LOW);

}
