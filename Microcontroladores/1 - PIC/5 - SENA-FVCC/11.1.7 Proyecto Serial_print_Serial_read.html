#include ".\Drivers\Arduino.h.encrypted"

#include ".\Drivers\LiquidCrystal.h"

 

#define boton_inc PIN_A1   //Boton incremento

#define boton_dec PIN_A2   //Boton decremento

#define boton_reset PIN_A3 //Boton reset

 

#define estado_boton_inc digitalRead(boton_inc)

#define estado_boton_dec digitalRead(boton_dec)

#define estado_boton_reset digitalRead(boton_reset)

 

int1 estado_boton_inc_ant;

int1 estado_boton_dec_ant;

 

int8 contador = 0;

float sensor;

 

char txt2stream[17];

char inputString[10];

int1 stringComplete;

 

void setup()

{

   pinMode(boton_inc,INPUT_);

   pinMode(boton_dec,INPUT_);

   pinMode(boton_reset,INPUT_);

   pinMode(PIN_B0, OUTPUT_);

   pinMode(PIN_B1, OUTPUT_);

   pinMode(PIN_B2, OUTPUT_);

   digitalWrite(LCD_BACKLIGHT, HIGH);

   lcd_begin(16,2);

   delay(200);

   lcd_setCursor(0,0);

   sprintf(txt2stream,"I: D: R: C:");

   lcd_print(txt2stream);

   lcd_setCursor(0,1);

   sprintf(txt2stream,"S:     L: M: B:");

   lcd_print(txt2stream);

   Serial_begin(9600);

}

 

void loop()

{

   if (estado_boton_inc!=estado_boton_inc_ant)

   {

      if (estado_boton_inc==1)

      {

         if (contador<100)

         {

            contador++;

         }

      }

      estado_boton_inc_ant = estado_boton_inc;

      delay(5);

   }

 

   sensor = analogRead(A0);

   sensor *= 100.0 / 1023.0;

 

   lcd_setCursor(2,0);

   sprintf(txt2stream,"%u",estado_boton_inc);

   lcd_print(txt2stream);

 

   lcd_setCursor(12,0);

   sprintf(txt2stream,"%03u",contador);

   lcd_print(txt2stream);

 

   lcd_setCursor(2,1);

   sprintf(txt2stream,"%.1f",sensor);

   lcd_print(txt2stream);

 

   sprintf(txt2stream,"$%u/%u/%u/%u/%.1f",estado_boton_inc,estado_boton_dec,estado_boton_reset,contador,sensor);

   Serial_println(txt2stream);

 

   if (stringComplete)

   {

      switch (inputString[0])

      {

         case '0': digitalWrite(PIN_B0, LOW); break;

         case '1': digitalWrite(PIN_B0, HIGH); break;

         default: break;

      }

 

      switch (inputString[1])

      {

         case '0': digitalWrite(PIN_B1, LOW); break;

         case '1': digitalWrite(PIN_B1, HIGH); break;

         default: break;

      }

 

      switch (inputString[2])

      {

         case '0': digitalWrite(PIN_B2, LOW); break;

         case '1': digitalWrite(PIN_B2, HIGH); break;

         default: break;

      }

 

      lcd_setCursor(9,1);

      sprintf(txt2stream,"%c",inputString[0]);

      lcd_print(txt2stream);

 

      // clear the string:

      sprintf(inputString,"");

      stringComplete = false;

   }

   delay(50);

}

void serialEvent()

{

   while (Serial_available())

   {

      char inChar[2] = "0";

      // get the new byte:

      inChar[0] = Serial_read();

      // add it to the inputString:

      strcat(inputString,inChar);

      // if the incoming character is a newline, set a flag

      // so the main loop can do something about it:

      if (inChar[0] == '\r')

      {

         stringComplete = true;

      }

   }

}

        void SerialPort1DataReceived(object sender, System.IO.Ports.SerialDataReceivedEventArgs e)
        {
            int cant_datos = 5;
            string datos;
            string[] vector_datos;
            int pulsador_inc;
            int pulsador_dec;
            int pulsador_reset;
            int contador;
            double sensor;

            if (serialPort1.BytesToRead > 1)
            {
                datos = serialPort1.ReadLine();
                if (datos.StartsWith("$"))
                {
                    vector_datos = datos.Substring(1).Split('/');
                    if (vector_datos.Length==cant_datos)
                    {
                        pulsador_inc = Convert.ToInt16(vector_datos[0]);
                        pulsador_dec = Convert.ToInt16(vector_datos[1]);
                        pulsador_reset = Convert.ToInt16(vector_datos[2]);
                        contador = Convert.ToInt16(vector_datos[3]);
                        sensor = Convert.ToDouble(vector_datos[4])/10;
                        label2.Text = pulsador_inc.ToString();
                        label5.Text = pulsador_dec.ToString();
                        label8.Text = pulsador_reset.ToString();
                        label11.Text = contador.ToString();
                        label13.Text = sensor.ToString();
                        if (pulsador_inc==1)
                            label3.BackColor = Color.LimeGreen;
                        else
                            label3.BackColor = Color.DarkGray;
                        if (pulsador_dec==1)
                            label6.BackColor = Color.LimeGreen;
                        else
                            label6.BackColor = Color.DarkGray;
                        if (pulsador_reset==1)
                            label9.BackColor = Color.LimeGreen;
                        else
                            label9.BackColor = Color.DarkGray;
                        trackBar1.Value = contador;
                        progressBar1.Value = Convert.ToInt16(sensor);
                    }
                }
            }
        }
