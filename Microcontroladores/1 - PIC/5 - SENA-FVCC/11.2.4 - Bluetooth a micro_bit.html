#include ".\Drivers\Arduino.h.encrypted"

#include ".\Drivers\LiquidCrystal.h"

 

char txt2stream[17];

char inputString[10];

int1 stringComplete;

 

void setup()

{

   pinMode(PIN_B0, OUTPUT_);

   pinMode(PIN_D3, OUTPUT_);

   digitalWrite(PIN_D3, HIGH);

   lcd_begin(16,2);

   delay(200);

   lcd_setCursor(0,0);

   sprintf(txt2stream,"LED:   M:   B:");

   lcd_print(txt2stream);

   Serial_begin(9600);

}

 

void loop()

{

   if (stringComplete)

   {

      switch (inputString[0])

      {

         case '0': digitalWrite(PIN_B0, LOW); break;

         case '1': digitalWrite(PIN_B0, HIGH); break;

         default: break;

      }

      switch (inputString[1])

      {

         case '0': digitalWrite(PIN_B0, LOW); break;

         case '1': digitalWrite(PIN_B0, HIGH); break;

         default: break;

      }

      lcd_setCursor(4,0);

      sprintf(txt2stream,"%c",inputString[0]);

      lcd_print(txt2stream);

      // clear the string:

      sprintf(inputString,"");

      stringComplete = false;

   }

}

 

#INT_RDA

void serialEvent()

{

   while (Serial_available())

   {

      char inChar[2];

      // get the new byte:

      inChar[0] = Serial_read();

      // add it to the inputString:

      strcat(inputString,inChar);

      // if the incoming character is a newline, set a flag

      // so the main loop can do something about it:

      if (inChar[0] == '\r')

      {

         stringComplete = true;

      }

   }

}
