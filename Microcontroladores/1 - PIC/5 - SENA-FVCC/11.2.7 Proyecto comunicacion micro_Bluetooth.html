#define estado_boton_inc digitalRead(boton_inc)

void setup() {

   pinMode(boton_inc,INPUT_);

   pinMode(boton_dec,INPUT_);

   pinMode(PIN_B1, OUTPUT_);

   pinMode(PIN_B2, OUTPUT_);

   lcd_begin(20,4);

   lcd_setCursor(0,0);

   sprintf(txt2stream,"I: D: R: C: ");

   lcd_print(txt2stream);

   lcd_setCursor(0,1);

   sprintf(txt2stream,"Sensor: ");

   lcd_print(txt2stream);

   lcd_setCursor(0,2);

   sprintf(txt2stream,"L: M: B: ");

   lcd_print(txt2stream);

   lcd_setCursor(0,3);

   sprintf(txt2stream,"PWM: ");

   lcd_print(txt2stream);

   Serial_begin(9600);

}

void serialEvent() {

   while (Serial_available())

   {

      char inChar[2]="0";

      // get the new byte:

      inChar[0] = Serial_read();

      // add it to the inputString:

      strcat(inputString,inChar);

      // if the incoming character is a newline, set a flag

      // so the main loop can do something about it:

      if (inChar[0] == '\r')

      {

         stringComplete = true;

      }

   }

}
