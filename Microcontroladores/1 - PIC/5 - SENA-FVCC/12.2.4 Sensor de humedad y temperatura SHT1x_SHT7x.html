Sensor de humedad y temperatura SHT1x_SHT7x

 

 

 

 

TECHNICAL DETAILS

 

Body dimensions: 14mm diameter, 50mm long

Cable length: 1 meter

Humidity readings with 4.5% accuracy

Temperature readings with 0.5 degree C accuracy

Working Temperature/Humidity range: -40°C ~ 120°C, 0~100% RH

Four wires: Red = VCC (3-5VDC), Black or Green = Ground, Yellow = Clock, Blue = Data

#include <Arduino.h.encrypted>

#include <LiquidCrystal.h>

 

#define sht_data_pin PIN_B2

#define sht_clk_pin PIN_B3

 

#include ".\Drivers\sht7x_sht1x.c"

 

float PV_Temp, PV_Hum;

char grados = 223;

char txt2stream[17];

 

void setup() {

   digitalWrite(LCD_BACKLIGHT, HIGH);

   lcd_begin(16,2);

   lcd_setCursor(0,0);

   sprintf(txt2stream,"PV_Temp:");

   lcd_print(txt2stream);

   lcd_setCursor(0,1);

   sprintf(txt2stream,"PV_Hum:");

   lcd_print(txt2stream);

   sht_init();

   Serial_begin(9600);

}

 

void loop() {

   sht_rd(PV_temp, PV_hum);

 

   lcd_setCursor(9,0);

   sprintf(txt2stream,"%.1f%cC ",PV_Temp,grados);

   lcd_print(txt2stream);

 

   lcd_setCursor(9,1);

   sprintf(txt2stream,"%.1f%% ",PV_Hum);

   lcd_print(txt2stream);

 

   sprintf(txt2stream,"$%.1f/%.1f",PV_Temp,PV_Hum);

   Serial_println(txt2stream);

   delay(50);

}
