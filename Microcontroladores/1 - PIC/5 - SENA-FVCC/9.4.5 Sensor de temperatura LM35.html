#include <Arduino.h.encrypted>

#include <LiquidCrystal.h>

 

int32 temperatura;

char grados = 223;

char txt2stream[10];

 

void setup() {

   digitalWrite(LCD_BACKLIGHT, HIGH);

   lcd_begin(16,2);

   delay(200);

   lcd_setCursor(0,0);

   sprintf(txt2stream,"Temp: ");

   lcd_print(txt2stream);

}

 

void loop() {

   temperatura = analogRead(A0);

   temperatura = temperatura * 150 / 1023;

   lcd_setCursor(6,0);

   sprintf(txt2stream,"%Lu %cC ",temperatura,grados);

   lcd_print(txt2stream);

   delay(500);

}
