#include ".\Drivers\Arduino.h.encrypted"

#include ".\Drivers\LiquidCrystal.h"

 

#define boton_inc PIN_A0 //Boton incremento

#define boton_dec PIN_A1 //Boton decremento

#define boton_reset PIN_A2 //Boton reset

 

#define estado_boton_inc digitalRead(boton_inc)

#define estado_boton_dec digitalRead(boton_dec)

#define estado_boton_reset digitalRead(boton_reset)

 

int1 estado_boton_inc_ant;

int1 estado_boton_dec_ant;

 

int8 contador = 0;

char txt2stream[17];

 

void setup()

{

   pinMode(PIN_A0, INPUT_);

   pinMode(PIN_A1, INPUT_);

   pinMode(PIN_A2, INPUT_);

   digitalWrite(LCD_BACKLIGHT, HIGH);

   lcd_begin(16,2);

   delay(200);

   lcd_setCursor(0,0);

   sprintf(txt2stream,"BI: BD: BR:");

   lcd_print(txt2stream);

   lcd_setCursor(0,1);

   sprintf(txt2stream,"Contador:");

   lcd_print(txt2stream);

}

 

void loop()

{

   if (estado_boton_inc!=estado_boton_inc_ant)

   {

      if ((estado_boton_inc==1)&&(contador<100))

      {

         contador++;

      }

      estado_boton_inc_ant = estado_boton_inc;

      delay(5);

   }

 

   if (estado_boton_dec!=estado_boton_dec_ant)

   {

      if ((estado_boton_dec==1)&&(contador>0))

      {

         contador--;

      }

      estado_boton_dec_ant = estado_boton_dec;

      delay(5);

   }

 

   if (estado_boton_reset==1)

   {

      contador = 0;

   }

 

   lcd_setCursor(3,0);

   sprintf(txt2stream,"%u",estado_boton_inc);

   lcd_print(txt2stream);

 

   lcd_setCursor(9,0);

   sprintf(txt2stream,"%u",estado_boton_dec);

   lcd_print(txt2stream);

 

   lcd_setCursor(15,0);

   sprintf(txt2stream,"%u",estado_boton_reset);

   lcd_print(txt2stream);

 

   lcd_setCursor(9,1);

   sprintf(txt2stream,"%u ",contador);

   lcd_print(txt2stream);

   delay(50);

}
