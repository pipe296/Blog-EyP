#include ".\Drivers\Arduino.h.encrypted"

 

int16 pins_PA[8] = {PIN_A0, PIN_A1, PIN_A2, PIN_A3, PIN_A4, PIN_A5, PIN_E0, PIN_E1};

int16 pins_PB[8] = {PIN_B0, PIN_B1, PIN_B2, PIN_B3, PIN_B4, PIN_B5, PIN_B6, PIN_B7};

 

int8 input_PA(){

   int8 cifra=0;

   int8 i;

   for (i=0; i<8; i++)

   {

      cifra=cifra+(digitalRead(pins_PA[i])*(1<<i));

   }

   return cifra;

}

 

void output_PB(int8 cifra){

   int8 i;

   for (i=0; i<8; i++)

   {

      digitalWrite(pins_PB[i], (cifra>>i)&1);

   }

}

 

void setup(){

   int8 i;

   for (i=0; i<8; i++)

   {

      pinMode(pins_PA[i], INPUT_);

      pinMode(pins_PB[i], OUTPUT_);

      digitalWrite(pins_PB[i], LOW);

   }

}

 

void loop(){

   output_PB(input_PA());

}
