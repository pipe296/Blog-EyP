#include ".\Drivers\Arduino.h.encrypted"

#include ".\Drivers\LiquidCrystal.h"

#include ".\Drivers\kbd_membrana.c.encrypted"

 

#define sht_data_pin PIN_B2

#define sht_clk_pin PIN_B3

 

#include ".\Drivers\sht7x_sht1x.c"

 

float SP_Temp, SP_Hum;

float PV_Temp, PV_Hum;

char grados = 223;

char tecla;

int8 estado=0;

char txt2stream[17];

 

void ver_menu()

{

   lcd_setCursor(0,0);

   sprintf(txt2stream,"A - Temperatura");

   lcd_print(txt2stream);

   lcd_setCursor(0,1);

   sprintf(txt2stream,"B - Humedad");

   lcd_print(txt2stream);

}

 

void ver_temp()

{

   lcd_setCursor(0,0);

   sprintf(txt2stream,"SP_Temp: %.1f%cC ",SP_Temp,grados);

   lcd_print(txt2stream);

   lcd_setCursor(0,1);

   sprintf(txt2stream,"PV_Temp: %.1f%cC ",PV_Temp,grados);

   lcd_print(txt2stream);

}

 

void ver_hum()

{

   lcd_setCursor(0,0);

   sprintf(txt2stream,"SP_Hum: %.1f%% ",SP_Hum);

   lcd_print(txt2stream);

   lcd_setCursor(0,1);

   sprintf(txt2stream,"PV_Hum: %.1f%% ",PV_Hum);

   lcd_print(txt2stream);

}

 

void setup()

{

   pinMode(PIN_D3, OUTPUT_);

   digitalWrite(PIN_D3, HIGH);

   lcd_begin(16,2);

   ver_menu();

   sht_init();

   kbd_init();

   Serial_begin(9600);

}

 

void loop()

{

   SP_temp = analogRead(A0);

   SP_temp = SP_temp * 100 / 1023;

   SP_Hum = analogRead(A1);

   SP_Hum = SP_Hum * 100 / 1023;

 

   sht_rd(PV_temp, PV_hum);

 

   // Escribir código de la aplicación

 

   tecla = kbd_getc();

 

   if (tecla=='*') {lcd_clear(); estado = 0;}

   if (tecla=='A') {lcd_clear(); estado = 1;}

   if (tecla=='B') {lcd_clear(); estado = 2;}

 

   if (estado==0) ver_menu();

   if (estado==1) ver_temp();

   if (estado==2) ver_hum();

 

   sprintf(txt2stream,"$%.1f/%.1f/%.1f/%.1f",SP_Temp,PV_Temp,SP_Hum,PV_Hum);

   Serial_println(txt2stream);

   delay(50);

}
