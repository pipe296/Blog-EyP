void setup() {

   pinMode(boton_inc,INPUT_);

   pinMode(boton_dec,INPUT_);

   pinMode(boton_reset,INPUT_);

   pinMode(PIN_B1, OUTPUT_);

   pinMode(PIN_B2, OUTPUT_);

   lcd_setCursor(0,2);

   lcd_print("L: M: B: ");

   lcd_setCursor(0,3);

   lcd_print("PWM: ");

}

      }

      switch (inputString[1])

      intPWM = atol(strPWM);

      floatPWM = intPWM / 10.0;

      analogWrite(PIN_C2,intPWM);

      lcd_print(txt2stream);

 

      // clear the string:

      sprintf(inputString,"");

      stringComplete = false;

   }

}

void serialEvent() {

   while (Serial_available())

   {

      char inChar[2]="0";

      // get the new byte:

      inChar[0] = Serial_read();

      // add it to the inputString:

      strcat(inputString,inChar);

      // if the incoming character is a newline, set a flag

      // so the main loop can do something about it:

      if (inChar[0] == '\r')

      {

         stringComplete = true;

      }

   }

}

        void SerialPort1DataReceived(object sender, System.IO.Ports.SerialDataReceivedEventArgs e)
        {
            int cant_datos = 5;
            string datos;
            string[] vector_datos;
            int pulsador_inc;
            int pulsador_dec;
            int pulsador_reset;
            int contador;
            double sensor;

            if (serialPort1.BytesToRead > 1)
            {
                datos = serialPort1.ReadLine();
                if (datos.StartsWith("$"))
                {
                    vector_datos = datos.Substring(1).Split('/');
                    if (vector_datos.Length==cant_datos)
                    {
                        pulsador_inc = Convert.ToInt16(vector_datos[0]);
                        label2.Text = pulsador_inc.ToString();
                        if (pulsador_inc==1)
                            label3.BackColor = Color.LimeGreen;
                        else
                            label3.BackColor = Color.DarkGray;

                        pulsador_dec = Convert.ToInt16(vector_datos[1]);
                        label5.Text = pulsador_dec.ToString();
                        if (pulsador_dec==1)
                            label6.BackColor = Color.LimeGreen;
                        else
                            label6.BackColor = Color.DarkGray;

                        //Programar acción pulsador_reset
                        pulsador_reset = Convert.ToInt16(vector_datos[2]);
                        label8.Text = pulsador_reset.ToString();
                        if (pulsador_reset==1)
                            label9.BackColor = Color.LimeGreen;
                        else
                            label9.BackColor = Color.DarkGray;

                        contador = Convert.ToInt16(vector_datos[3]);
                        label11.Text = contador.ToString();
                        trackBar1.Value = contador;

                        sensor = Convert.ToDouble(vector_datos[4])/10;
                        label13.Text = sensor.ToString();
                        progressBar1.Value = Convert.ToInt16(sensor);
                    }
                }
            }
        }
