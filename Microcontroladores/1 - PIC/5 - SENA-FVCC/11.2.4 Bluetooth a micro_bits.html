#include <Arduino.h.encrypted>

#include <LiquidCrystal.h>

 

char txt2stream[17];

char inputString[10];

int1 stringComplete;

 

void setup() {

   pinMode(PIN_B0, OUTPUT_);

   pinMode(PIN_B1, OUTPUT_);

   digitalWrite(LCD_BACKLIGHT, HIGH);

   lcd_begin(16,2);

   delay(200);

   lcd_setCursor(0,0);

   sprintf(txt2stream,"LED:   M:   B:");

   lcd_print(txt2stream);

   Serial_begin(9600);

}

 

void loop() {

   if (stringComplete)

   {

      switch (inputString[0])

      {

         case '0': digitalWrite(PIN_B0, LOW); break;

         case '1': digitalWrite(PIN_B0, HIGH); break;

         default: break;

      }

      switch (inputString[1])

      {

         case '0': digitalWrite(PIN_B1, LOW); break;

         case '1': digitalWrite(PIN_B1, HIGH); break;

         default: break;

      }

      

      //Programar acción inputString[2]

      

      lcd_setCursor(4,0);

      sprintf(txt2stream,"%c",inputString[0]);

      lcd_print(txt2stream);

      

      lcd_setCursor(9,0);

      sprintf(txt2stream,"%c",inputString[1]);

      lcd_print(txt2stream);

      

      //Programar visualización inputString[2]

      

      // clear the string:

      sprintf(inputString,"");

      stringComplete = false;

   }

}

 

#INT_RDA

void serialEvent() {

   while (Serial_available())

   {

      char inChar[2] = "0";

      // get the new byte:

      inChar[0] = Serial_read();

      // add it to the inputString:

      strcat(inputString,inChar);

      // if the incoming character is a newline, set a flag

      // so the main loop can do something about it:

      if (inChar[0] == '\r')

      {

         stringComplete = true;

      }

   }

}
