#include <Arduino.h.encrypted>

#include <LiquidCrystal.h>

 

int sensor;

 

char txt2stream[10];

 

void setup() {

   digitalWrite(LCD_BACKLIGHT, HIGH);

   lcd_begin(16,2);

   delay(200);

   lcd_setCursor(0,0);

   lcd_print("Sensor:");

}

 

void loop() {

   sensor = analogRead(A0);

   sensor = map(sensor,0,1023,0,100);

   lcd_setCursor(8,0);

   sprintf(txt2stream,"%Lu   ",sensor);

   lcd_print(txt2stream);

   delay(500);

}
