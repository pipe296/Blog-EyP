#include <Arduino.h.encrypted>

#include <LiquidCrystal.h>

#include ".\Drivers\DS1307.c"

 

int8 sec;

int8 min;

int8 hour;

int8 day;

int8 month;

int8 year;

int8 dow;

char sdow[10];

char cmd;

char txt2stream[20];

 

void leer_fecha_y_hora() {

   ds1307_get_day_of_week((char*) sdow);

   ds1307_get_date(day,month,year,dow);

   ds1307_get_time(hour,min,sec);

   lcd_setCursor(0,0);

   sprintf(txt2stream,"%s %02d/%02d/%02d ",sdow,day,month,year);

   lcd_print(txt2stream);

   lcd_setCursor(0,1);

   sprintf(txt2stream,"%02d:%02d:%02d",hour,min,sec);

   lcd_print(txt2stream);

}

 

void setup() {

   pinMode(PIN_B2, OUTPUT_);

   digitalWrite(LCD_BACKLIGHT, HIGH);

   lcd_begin(16,2);

   delay(200);

   // Inicializa DS1307

   ds1307_init(DS1307_ALL_DISABLED);

   ds1307_set_date_time(30,5,18,2,5,59,55);

}

 

void loop() {

   leer_fecha_y_hora();

   if ((dow>=0)&&(dow<=5)) //Solamente de Lunes (dow=0) a Sábado (dow=5)

   {

      if ((hour==6)&&(min==0)&&(sec<5)) //Durante 5 segundos a las 6:00

      {

         digitalWrite(PIN_B2, HIGH);

         delay(500);

         digitalWrite(PIN_B2, LOW);

         sprintf(txt2stream,"Alarma!");

         lcd_print(txt2stream);

      }

      else if ((hour==14)&&(min==0)&&(sec<5)) //Durante 5 segundos a las 14:00

      {

         digitalWrite(PIN_B2, HIGH);

         delay(500);

         digitalWrite(PIN_B2, LOW);

         sprintf(txt2stream,"Alarma!");

         lcd_print(txt2stream);

      }

      else

      {

         lcd_setCursor(9,1);

         sprintf(txt2stream," ");

         lcd_print(txt2stream);

      }

   }

   delay(500);

}
