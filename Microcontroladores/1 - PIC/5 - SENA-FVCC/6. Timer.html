#include ".\Drivers\Arduino.h.encrypted"

#include ".\Drivers\LiquidCrystal.h"

 

#define pul_sel PIN_A0

#define pul_inc PIN_A1

#define pul_dec PIN_A2

#define pul_start PIN_A3

#define pul_stop PIN_A4

 

#define estado_pul_sel digitalRead(pul_sel)

#define estado_pul_inc digitalRead(pul_inc)

#define estado_pul_dec digitalRead(pul_dec)

#define estado_pul_start digitalRead(pul_start)

#define estado_pul_stop digitalRead(pul_stop)

 

int1 temporizar=0;

int8 min=0;

int8 seg=0;

int8 menu=0;

char txt2stream[17];

 

void setup() {

   pinMode(PIN_A0,INPUT_);

   pinMode(PIN_A1,INPUT_);

   pinMode(PIN_A2,INPUT_);

   pinMode(PIN_A3,INPUT_);

   pinMode(PIN_A4,INPUT_);

   digitalWrite(LCD_BACKLIGHT,HIGH);

   lcd_begin(16,2);

   delay(200);

   lcd_setCursor(0,0);

   sprintf(txt2stream,"Tiempo 00m 00s");

   lcd_print(txt2stream);

}

 

void loop() {

   if (estado_pul_sel==1)

   {

      if (menu==0)

      {

         lcd_setCursor(8,0);

         lcd_blink();

         menu=1;

      }

      else if (menu==1)

      {

         lcd_setCursor(12,0);

         lcd_blink();

         menu=2;

      }

      else if (menu==2)

      {

         lcd_noblink();

         menu=0;

      }

      while (estado_pul_sel==1)

      {

         delay(10);

      }

   }

 

   if (estado_pul_inc==1)

   {

      while (estado_pul_inc==1)

      {

         if (menu==1)

         {

            min++;

            if (min>=60)

               min=0;

            sprintf(txt2stream,"%02um",min);

            lcd_setCursor(7,0);

            lcd_print(txt2stream);

            lcd_setCursor(8,0);

         }

         if (menu==2)

         {

            seg++;

            if (seg>=60)

               seg=0;

            sprintf(txt2stream,"%02us",seg);

            lcd_setCursor(11,0);

            lcd_print(txt2stream);

            lcd_setCursor(12,0);

         }

         delay(500);

      }

   }

 

   if (estado_pul_dec==1)

   {

      while (estado_pul_dec==1)

      {

         if (menu==1)

         {

            if (min>0)

               min--;

            sprintf(txt2stream,"%02um",min);

            lcd_setCursor(7,0);

            lcd_print(txt2stream);

            lcd_setCursor(8,0);

         }

         if (menu==2)

         {

            if (seg>0)

               seg--;

            sprintf(txt2stream,"%02us",seg);

            lcd_setCursor(11,0);

            lcd_print(txt2stream);

            lcd_setCursor(12,0);

         }

         delay(500);

      }

   }

 

   if (estado_pul_start==1 && menu==0)

   {

      temporizar=1;

   }

 

   if (estado_pul_stop==1)

   {

      if (menu==0)

      {

         temporizar=0;

      }

      if (menu==1)

      {

         min = 0;

         sprintf(txt2stream,"%02um",min);

         lcd_setCursor(7,0);

         lcd_print(txt2stream);

         lcd_setCursor(8,0);

      }

      if (menu==2)

      {

         seg = 0;

         sprintf(txt2stream,"%02us",seg);

         lcd_setCursor(11,0);

         lcd_print(txt2stream);

         lcd_setCursor(12,0);

      }

   }

 

   if (temporizar==1)

   {

      delay(1000); //Periodo de temporización

      if (seg>0)

         seg--;

      else if ((seg==0)&&(min>0))

      {

         seg=59;

         min--;

      }

      if ((seg==0)&&(min==0))

      {

         temporizar=0;

         analogWrite(C1,0);

      }

      sprintf(txt2stream,"%02um",min);

      lcd_setCursor(7,0);

      lcd_print(txt2stream);

      sprintf(txt2stream,"%02us",seg);

      lcd_setCursor(11,0);

      lcd_print(txt2stream);

   }

}
