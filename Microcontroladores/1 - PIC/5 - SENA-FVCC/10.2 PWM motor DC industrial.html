#include <Arduino.h.encrypted>

#include ".\Drivers\LiquidCrystal.h"

 

int32 sensor = 0;

char txt2stream[12];

 

void setup() {

   digitalWrite(LCD_BACKLIGHT, HIGH);

   lcd_begin(16,2);

   delay(200);

   sprintf(txt2stream,"PWM: 0");

   lcd_setCursor(0,0);

   lcd_print(txt2stream);

}

 

void loop() {

   sensor = analogRead(A0);

   sensor = sensor * 1000 / 1023;

   sprintf(txt2stream,"%Lu ",sensor);

   lcd_setCursor(5,0);

   lcd_print(txt2stream);

   analogWrite(PIN_C2,sensor);

   delay(50);

}
