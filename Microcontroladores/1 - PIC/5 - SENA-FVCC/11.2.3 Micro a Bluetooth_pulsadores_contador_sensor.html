#include ".\Drivers\Arduino.h.encrypted"

#include ".\Drivers\LiquidCrystal.h"

 

#define boton_inc PIN_A1   //Boton incremento

#define boton_dec PIN_A2   //Boton decremento

#define boton_reset PIN_A3 //Boton reset

 

#define estado_boton_inc digitalRead(boton_inc)

#define estado_boton_dec digitalRead(boton_dec)

#define estado_boton_reset digitalRead(boton_reset)

 

int1 estado_boton_inc_ant;

int1 estado_boton_dec_ant;

 

int8 contador = 0;

float sensor;

 

char txt2stream[17];

 

void setup()

{

   pinMode(boton_inc,INPUT_);

   pinMode(boton_dec,INPUT_);

   pinMode(boton_reset,INPUT_);

   digitalWrite(LCD_BACKLIGHT, HIGH);

   lcd_begin(16,2);

   delay(200);

   lcd_setCursor(0,0);

   sprintf(txt2stream,"BI:   BD:   BR: ");

   lcd_print(txt2stream);

   lcd_setCursor(0,1);

   sprintf(txt2stream,"Con:    Sen:");

   lcd_print(txt2stream);

   Serial_begin(9600);

}

 

void loop()

{

   if (estado_boton_inc!=estado_boton_inc_ant)

   {

      if (estado_boton_inc==1)

      {

         if (contador<100)

         {

            contador++;

         }

      }

      estado_boton_inc_ant = estado_boton_inc;

      delay(5);

   }

 

   //Programar acción boton_dec

 

   //Programar acción boton_reset

 

   sensor = analogRead(A0);

   sensor *= 100.0 / 1023.0;

 

   lcd_setCursor(3,0);

   sprintf(txt2stream,"%u",estado_boton_inc);

   lcd_print(txt2stream);

 

   //Programar visualización estado_boton_dec

 

   //Programar visualización estado_boton_reset

   

   lcd_setCursor(4,1);

   sprintf(txt2stream,"%u",contador);

   lcd_print(txt2stream);

 

   lcd_setCursor(12,1);

   sprintf(txt2stream,"%.1f",sensor);

   lcd_print(txt2stream);

 

   sprintf(txt2stream,"$%u/%u/%u/%u/%.1f",estado_boton_inc,estado_boton_dec,estado_boton_reset,contador,sensor);

   Serial_println(txt2stream);

   delay(50);

}
