#include ".\Drivers\Arduino.h.encrypted"

#include ".\Drivers\LiquidCrystal.h"

#include ".\Drivers\DS1307.c"

 

#define boton PIN_A4

 

int1 estado_boton;

 

int8 yr;

int8 month;

int8 day;

int8 hrs;

int8 sec;

int8 min;

int8 dow;

char sdow[11];

 

char txt2stream[17];

 

void leer_fecha_y_hora(void)

{

   ds1307_get_day_of_week((char*) sdow);

   ds1307_get_date(day,month,yr,dow);

   ds1307_get_time(hrs,min,sec);

   lcd_setCursor(0,0);

   sprintf(txt2stream,"\%s \%02d/\%02d/\%02d ",sdow,day,month,yr);

   lcd_print(txt2stream);

   lcd_setCursor(0,1);

   sprintf(txt2stream,"\%02d:\%02d:\%02d",hrs,min,sec);

   lcd_print(txt2stream);

}

 

void setup()

{

   pinMode(PIN_D3, OUTPUT_);

   digitalWrite(PIN_D3, HIGH);

   lcd_begin(16,2);

   delay(200);

   lcd_setCursor(5,0);

   sprintf(txt2stream,"Nombre");

   lcd_print(txt2stream);

   lcd_setCursor(4,1);

   sprintf(txt2stream,"proyecto");

   lcd_print(txt2stream);

   delay(1500);

   lcd_clear();

   lcd_setCursor(2,0);

   sprintf(txt2stream,"Ficha:xxxxxxx");

   lcd_print(txt2stream);

   lcd_setCursor(0,1);

   sprintf(txt2stream,"SENA Reg. Caldas");

   lcd_print(txt2stream);

   delay(1500);

   lcd_clear();

   // Inicializa DS1307

   ds1307_init(DS1307_ALL_DISABLED);

   // Configura Fecha y Hora actual y transmite

   // Set date for -> 15 Septiembre 2017 Viernes

   // Set time for -> 08:30:00

   // ds1307_set_date_time(15,9,17,4,8,30,00);

}

 

void loop()

{

   // Escribir código de la aplicación

 

   estado_boton = digitalRead(boton);

 

   if (estado_boton==1)

   {

      while (estado_boton==1)

      {

         leer_fecha_y_hora();

         delay(1000);

         estado_boton = digitalRead(boton);

      }

      lcd_clear();

   }

}
