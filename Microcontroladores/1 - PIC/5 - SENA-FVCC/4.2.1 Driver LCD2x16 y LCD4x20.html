#include ".\Drivers\Arduino.h.encrypted"

#include ".\Drivers\LiquidCrystal.h"

 

char txt2stream[10];

 

void setup() {

   pinMode(PIN_D3, OUTPUT_);

   digitalWrite(PIN_D3, HIGH);

   lcd_begin(20,4);

   delay(200);

}

 

void loop() {

   lcd_setCursor(0,0);

   sprintf(txt2stream,"Linea 1");

   lcd_print(txt2stream);

   delay(1000);

   lcd_setCursor(0,1);

   sprintf(txt2stream,"Linea 2");

   lcd_print(txt2stream);

   delay(1000);

   lcd_setCursor(0,2);

   sprintf(txt2stream,"Linea 3");

   lcd_print(txt2stream);

   delay(1000);

   lcd_setCursor(0,3);

   sprintf(txt2stream,"Linea 4");

   lcd_print(txt2stream);

   delay(1000);

 

   // clear screen:

   lcd_clear();

 

   int8 thisChar;

   // print from 0 to 9:

   for (thisChar = 0; thisChar < 10; thisChar++) {

      sprintf(txt2stream,"%u",thisChar);

      lcd_print(txt2stream);

      delay(500);

   }

  

   // set the cursor to (19,1):

   lcd_setCursor(20, 1);

   // set the display to automatically scroll:

   lcd_autoscroll();

   // print from 0 to 9:

   for (thisChar = 0; thisChar < 10; thisChar++) {

      sprintf(txt2stream,"%u",thisChar);

      lcd_print(txt2stream);

      delay(500);

   }

  

   // turn off automatic scrolling

   lcd_noAutoscroll();

   delay(1000);

 

   // clear screen:

   lcd_clear();

 

   // Print a message to the LCD.

   lcd_print("blink");

   // Turn off the blinking cursor:

   lcd_blink();

   delay(2000);

   // clear screen for the next loop:

   lcd_clear();

   // Print a message to the LCD.

   lcd_print("noBlink");

   // Turn on the blinking cursor:

   lcd_noBlink();

   delay(2000);

 

   // clear screen:

   lcd_clear();

 

   // Print a message to the LCD.

   lcd_print("cursor");

   // Turn on the blinking cursor:

   lcd_cursor();

   delay(2000);

   // clear screen for the next loop:

   lcd_clear();

   // Print a message to the LCD.

   lcd_print("noCursor");

   // Turn off the blinking cursor:

   lcd_noCursor();

   delay(2000);

 

   // clear screen:

   lcd_clear();

 

   // Print a message to the LCD.

   lcd_print("display");

   // Turn on the display:

   lcd_display();

   delay(2000);

   // Turn off the display:

   lcd_noDisplay();

   delay(2000);

   lcd_display();

   delay(2000);

 

   // clear screen:

   lcd_clear();

 

   lcd_setCursor(12,0);

   // Print a message to the LCD.

   lcd_print("Izquierda");

 

   int8 positionCounter;

   // scroll 12 positions (string length) to the left

   // to move it offscreen left:

   for (positionCounter = 0; positionCounter < 12; positionCounter++) {

      // scroll one position left:

      lcd_scrollDisplayLeft();

      // wait a bit:

      delay(150);

   }

 

   delay(1000);

   // set the cursor to (0,0):

   lcd_home();

   // clear screen:

   lcd_clear();

 

   // Print a message to the LCD.

   lcd_print("Derecha");

   // scroll 13 positions (string length + display length) to the right

   // to move it offscreen right:

   for (positionCounter = 0; positionCounter < 13; positionCounter++) {

      // scroll one position right:

      lcd_scrollDisplayRight();

      // wait a bit:

      delay(150);

   }

 

   delay(1000);

   // clear screen:

   lcd_clear();

 

   lcd_leftToRight();

   for (thisChar = 65; thisChar < 85; thisChar++) {

      sprintf(txt2stream,"%c",thisChar);

      // print the character

      lcd_print(txt2stream);

      delay(300);

   }

 

   lcd_setCursor(19,0);

   lcd_rightToLeft();

   for (thisChar = 97; thisChar < 117; thisChar++) {

      sprintf(txt2stream,"%c",thisChar);

      // print the character

      lcd_print(txt2stream);

      delay(300);

   }

 

   delay(1000);

   // clear screen:

   lcd_clear();

}
