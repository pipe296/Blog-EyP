#include <Arduino.h.encrypted>

#include <LiquidCrystal.h>

 

char inputString[10];

int1 stringComplete;

int16 intPWM = 0;

float floatPWM = 0;

char txt2stream[12];

 

void setup() {

   pinMode(PIN_C2, OUTPUT_);

   digitalWrite(LCD_BACKLIGHT, HIGH);

   lcd_begin(16,2);

   delay(200);

   lcd_setCursor(0,0);

   sprintf(txt2stream,"PWM: 0%%");

   lcd_print(txt2stream);

   Serial_begin(9600);

}

 

void loop() {

   if (stringComplete)

   {

      intPWM = atol(inputString);

      floatPWM = intPWM / 10.0;

      sprintf(txt2stream,"%3.1f%% ",floatPWM);

      lcd_setCursor(5,0);

      lcd_print(txt2stream);

      analogWrite(PIN_C2,intPWM);

      // clear the string:

      sprintf(inputString,"");

      stringComplete = false;

   }

}

 

#INT_RDA

void serialEvent() {

   while (Serial_available())

   {

      char inChar[2] = "0";

      // get the new byte:

      inChar[0] = Serial_read();

      // add it to the inputString:

      strcat(inputString,inChar);

      // if the incoming character is a newline, set a flag

      // so the main loop can do something about it:

      if (inChar[0] == '\r')

      {

         stringComplete = true;

      }

   }

}
