#include <Arduino.h.encrypted>

 

#define boton_inc PIN_A0      //Boton incremento

#define boton_dec PIN_A1      //Boton decremento

#define boton_reset PIN_A2    //Boton reset

 

#define estado_boton_inc digitalRead(boton_inc)

#define estado_boton_dec digitalRead(boton_dec)

#define estado_boton_reset digitalRead(boton_reset)

 

boolean estado_boton_inc_ant;

boolean estado_boton_dec_ant;

 

byte nidades = 0;

byte decenas = 0;

 

int pins_PB[8] = {PIN_B0, PIN_B1, PIN_B2, PIN_B3, PIN_B4, PIN_B5, PIN_B6, PIN_B7};

 

void write_digit(byte cifra) {

   byte i;

   for (i=0; i<8; i++)

   {

      digitalWrite(pins_PB[i], (cifra>>i)&1);

   }

}

 

void setup() {

   byte i;

   pinMode(boton_inc, INPUT_);

   pinMode(boton_dec, INPUT_);

   pinMode(boton_reset, INPUT_);

   for (i=0; i<8; i++)

   {

      pinMode(pins_PB[i], OUTPUT_);

   }

}

 

void loop() {

   if (estado_boton_inc!=estado_boton_inc_ant)

   {

      if ((estado_boton_inc==1)&&(decenas+unidades<153))

      {

         unidades++;

         if (unidades>9)

         {

            unidades = 0;

            decenas += 16;

         }

      }

      estado_boton_inc_ant = estado_boton_inc;

      delay(5);

   }

 

   //Programar acción boton_dec

 

   if (estado_boton_reset==1)

   {

      unidades = 0;

      decenas = 0;

   }

 

   write_digit(decenas + unidades);

}
