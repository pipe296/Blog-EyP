#include <Arduino.h.encrypted>

 

byte unidades = 0;

 

int pins_PB[8] = {PIN_B0, PIN_B1, PIN_B2, PIN_B3, PIN_B4, PIN_B5, PIN_B6, PIN_B7};

 

void output_PB(byte cifra) {

   byte i;

   for (i=0; i<8; i++)

   {

      digitalWrite(pins_PB[i], (cifra>>i)&1);

   }

}

 

void setup() {

   byte i;

   for (i=0; i<8; i++)

   {

      pinMode(pins_PB[i], OUTPUT_);

      digitalWrite(pins_PB[i], LOW);

   }

}

 

void loop() {

   output_PB(unidades);

   unidades++;

   delay(1000);

}
