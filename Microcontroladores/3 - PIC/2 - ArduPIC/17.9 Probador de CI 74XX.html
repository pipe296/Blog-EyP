#include <Arduino.h.encrypted>

#include <LiquidCrystal.h>

 

char inputString[10];

boolean stringComplete;

boolean a = 0;

boolean b = 0;

boolean c = 0;

boolean d = 0;

 

void setup() {

   pinMode(PIN_B0, INPUT_);

   pinMode(PIN_B1, OUTPUT_);

   pinMode(PIN_B2, OUTPUT_);

   Serial_begin(9600);

}

 

void loop() {

   if (stringComplete)

   {

      switch (inputString[0])

      {

         case 'T': digitalWrite(PIN_B1, LOW); digitalWrite(PIN_B2, LOW);

         delay(50);

         a = digitalRead(PIN_B0);

         delay(50);

         digitalWrite(PIN_B1, LOW); digitalWrite(PIN_B2, HIGH);

         delay(50);

         b = digitalRead(PIN_B0);

         delay(50);

         digitalWrite(PIN_B1, HIGH); digitalWrite(PIN_B2, LOW);

         delay(50);

         c = digitalRead(PIN_B0);

         delay(50);

         digitalWrite(PIN_B1, HIGH); digitalWrite(PIN_B2, HIGH);

         delay(50);

         d = digitalRead(PIN_B0);

         digitalWrite(PIN_B1, LOW); digitalWrite(PIN_B2, LOW);

         if (a==0 && b==0 && c==0 && d==1)

            Serial_println("AB"); //AND Buena

         else if (a==0 && b==1 && c==1 && d==1)

            Serial_println("OB"); //OR Buena

         else if (a==1 && b==1 && c==1 && d==0)

            Serial_println("DB"); //NAND Buena

         else if (a==1 && b==0 && c==0 && d==0)

            Serial_println("RB"); //NOR Buena

         else if (a==0 && b==1 && c==1 && d==0)

            Serial_println("XB"); //XOR Buena

         else

            Serial_println("M");

         break;

         default: break;

      }

 

      // clear the string:

      sprintf(inputString,"");

      stringComplete = false;

   }

}

 

#INT_RDA

void serialEvent() {

   while (Serial_available())

   {

      char inChar[2]="0";

      // get the new byte:

      inChar[0] = Serial_read();

      // add it to the inputString:

      strcat(inputString,inChar);

      // if the incoming character is a newline, set a flag

      // so the main loop can do something about it:

      if (inChar[0] == '\r')

      {

         stringComplete = true;

      }

   }

}

        void Button_BuscarClick(object sender, EventArgs e)
        {
            comboBox_Puertos.Items.Clear();

                serialPort1.Open();
                button_Conectar.Enabled = false;
                button1.Enabled = true;
            }
