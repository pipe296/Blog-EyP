#include <16F877A.h>

//#include <18F4550.h>

 

#fuses HS,NOWDT,NOPROTECT,NOLVP,NODEBUG

#use delay(clock=20000000)

 

void main()

{

  while (true) //Bucle infinito que mantendrá el programa en ejecución

  {

    if (!INPUT(PIN_A0)==1) //Lee la entrada en el pin A0 y determina si está activa o inactiva

      OUTPUT_HIGH(PIN_B0); //Si la entrada A0 está inactiva (conectada a 0V) se activa la salida B0

    else

      OUTPUT_LOW(PIN_B0); //Si la entrada A0 está activa (conectada a 5V) se desactiva la salida B0

  }

}
