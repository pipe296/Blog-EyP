#include <LiquidCrystal.h>

 

LiquidCrystal lcd(P2_0, P2_1, P2_2, P2_3, P2_4, P2_5);

 

#define boton_inc P1_0   //Boton incremento

#define boton_dec P1_1   //Boton decremento

#define boton_reset P1_2 //Boton reset

 

boolean estado_boton_inc;

boolean estado_boton_dec;

boolean estado_boton_reset;

 

boolean estado_boton_inc_ant;

boolean estado_boton_dec_ant;

 

byte contador = 0;

 

char txt2stream[12];

 

void setup()

{

  pinMode(boton_inc, INPUT_PULLUP);

  pinMode(boton_dec, INPUT_PULLUP);

  pinMode(boton_reset, INPUT_PULLUP);

  lcd.begin(16, 2);

  delay(200);

  lcd.setCursor(0,0);

  sprintf(txt2stream,"BI:   BD:   BR:");

  lcd.print(txt2stream);

  lcd.setCursor(0,1);

  sprintf(txt2stream,"Contador:");

  lcd.print(txt2stream);

}

 

void loop()

{

  estado_boton_inc = digitalRead(boton_inc);

  estado_boton_reset = digitalRead(boton_reset);

  if (estado_boton_inc!=estado_boton_inc_ant)

  {

    if ((estado_boton_inc==0)&&(contador<100))

    {

      contador++;

    }

    estado_boton_inc_ant = estado_boton_inc;

    delay(5);

  }

 

  //Programar acción boton_dec

 

  if (estado_boton_reset==0)

  {

    contador = 0;

  }

 

  lcd.setCursor(3,0);

  sprintf(txt2stream,"%u",contador);

  lcd.print(txt2stream);

  lcd.setCursor(10,1);

  sprintf(txt2stream,"%u ",contador);

  lcd.print(txt2stream);

  delay(50);

}
