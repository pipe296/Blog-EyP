Entrada digital puerto

 

int pins_P1[8] = {P1_0, P1_1, P1_2, P1_3, P1_4, P1_5, P1_6, P1_7};  
int pins_P2[8] = {P2_0, P2_1, P2_2, P2_3, P2_4, P2_5, P2_6, P2_7};

byte input_P1(){
  byte cifra=0;
  byte i;
  for (i=0; i<8; i++)
  {
    cifra=cifra+(digitalRead(pins_P1[i])*1<<i);
  }
  return cifra;
}

void output_P2(byte cifra){
  byte i;
  for (i=0; i<8; i++)
  {
    digitalWrite(pins_P2[i], cifra&1<<i);
  }
}
     
void setup(){
  byte i;
  for(i=0; i<8; i++)
  {
    pinMode(pins_P1[i], INPUT_PULLDOWN);
    pinMode(pins_P2[i], OUTPUT);
    digitalWrite(pins_P2[i], LOW);
  }
}
     
void loop(){
  output_P2(input_P1());
}
