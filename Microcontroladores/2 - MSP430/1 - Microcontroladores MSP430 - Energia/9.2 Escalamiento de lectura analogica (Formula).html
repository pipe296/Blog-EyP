Escalamiento de lectura analógica (Fórmula)

 

 

En el siguiente ejemplo haremos que el programa lea una entrada entre 0 y 5V de manera que para 0V representará 0 unidades y 5V representará 100 unidades.

 

0 a 100

 

// include the library code:

#include <LiquidCrystal.h>

 

// initialize the library with the numbers of the interface pins

LiquidCrystal lcd(P2_0, P2_1, P2_2, P2_3, P2_4, P2_5);

 

float sensor;

 

char txt2stream[10];

 

void setup()

{

  lcd.begin(16, 2);

  delay(200);

  lcd.setCursor(0,0);

  lcd.print("Sensor:");

}

 

void loop()

{

  sensor = analogRead(A0);

  sensor = sensor * 100 / 1023;

  dostrf(sensor,3,2,txt2stream);

  lcd.setCursor(8,0);

  lcd.print(txt2stream);

  delay(500);

}

   sensor = analogRead(A0);

   sprintf(txt2stream,"%Lu   ",sensor);

   lcd_print(txt2stream);

}
