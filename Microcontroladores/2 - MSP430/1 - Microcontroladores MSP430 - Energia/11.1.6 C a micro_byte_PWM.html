// include the library code:

#include <LiquidCrystal.h>

 

// initialize the library with the numbers of the interface pins

LiquidCrystal lcd(P2_0, P2_1, P2_2, P2_3, P2_4, P2_5);

 

String inputString;

boolean stringComplete;

int intPWM = 0;

float floatPWM = 0;

char txt2stream[12];

 

void setup()

{

  pinMode(P1_6, OUTPUT);

  lcd.begin(16,2);

  delay(200);

  lcd.setCursor(0,0);

  sprintf(txt2stream,"PWM: 0%%");

  lcd.print(txt2stream);

  Serial.begin(9600);

}

 

void loop()

{

  if (stringComplete)

  {

    intPWM = inputString.toInt();

    floatPWM = intPWM / 10.0;

    lcd.setCursor(5,0);

    lcd.print(floatPWM);

    lcd.print("% ");

    intPWM = map(intPWM,0,1000,0,255);

    analogWrite(P1_6,intPWM);

    // clear the string:

    inputString="";

    stringComplete = false;

  }

}

 

void serialEvent()

{

  while (Serial.available())

  {

    char inChar[2];

    // get the new byte:

    inChar[0] = Serial.read();

    // add it to the inputString:

    inputString += inChar[0];

    // if the incoming character is a newline, set a flag

    // so the main loop can do something about it:

    if (inChar[0] == '\r')

    {

      stringComplete = true;

    }

  }

}
