Salida digital puerto

 

byte unidades = 0;

 

byte pins_P2[8] = {P2_0, P2_1, P2_2, P2_3, P2_4, P2_5, P2_6, P2_7};

 

void output_P2(byte cifra){

  byte i;

  for (i=0; i<8; i++)

  {

    digitalWrite(pins_P2[i], (cifra>>i)&1);

  }

}

 

void setup(){

  byte i;

  for (i=0; i<8; i++)

  {

    pinMode(pins_P2[i], OUTPUT);

    digitalWrite(pins_P2[i], LOW);

  }

}

 

void loop(){

  output_P2(unidades);

  unidades++;

  delay(1000);

}
