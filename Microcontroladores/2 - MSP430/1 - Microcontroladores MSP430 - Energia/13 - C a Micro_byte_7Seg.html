String inputString;

boolean stringComplete;

 

byte pins_P2[8] = {P2_0, P2_1, P2_2, P2_3, P2_4, P2_5, P2_6, P2_7};

 

void output_P2(byte cifra){

  byte i;

  for (i=0; i<8; i++)

  {

    digitalWrite(pins_P2[i], (cifra>>i)&1);

  }

}

 

void setup()

{

  byte i;

  for (i=0; i<8; i++)

  {

    pinMode(pins_P2[i], OUTPUT);

    digitalWrite(pins_P2[i], LOW);

  }

  Serial.begin(9600);

}

 

void loop()

{

  if (stringComplete)

  {

    switch (inputString[0])

    {

      case '0': output_P2(0b00111111); break;

      case '1': output_P2(6); break;

      case '2': output_P2(91); break;

      case '3': output_P2(79); break;

      case '4': output_P2(102); break;

      case 53: output_P2(109); break;

      case 54: output_P2(125); break;

      case 55: output_P2(7); break;

      case 56: output_P2(127); break;

      case 57: output_P2(103); break;

      // Escribir código para procesar A-F

      default: break;

    }

    // clear the string:

    inputString="";

    stringComplete = false;

  }

}

 

void serialEvent()

{

  while (Serial.available())

  {

    char inChar[2];

    // get the new byte:

    inChar[0] = Serial.read();

    // add it to the inputString:

    inputString += inChar[0];

    // if the incoming character is a newline, set a flag

    // so the main loop can do something about it:

    if (inChar[0] == '\r')

    {

      stringComplete = true;

    }

  }

}
