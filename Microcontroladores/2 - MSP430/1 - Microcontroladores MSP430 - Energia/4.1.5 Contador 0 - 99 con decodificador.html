/**

Display 7 segmenti per corso arduino.

segmenti:

a P2_0

b P2_1

c P2_2

d P2_3

e P2_4

f P2_5

g P2_6

*/

 

#define boton_inc P1_3   //Boton incremento

#define boton_dec P1_4   //Boton decremento

#define boton_reset P1_5 //Boton reset

 

boolean estado_boton_inc;

boolean estado_boton_dec;

boolean estado_boton_reset;

 

boolean estado_boton_inc_ant;

boolean estado_boton_dec_ant;

 

byte unidades = 0;

byte decenas = 0;

 

long segment_to_pin[8] = {P2_0, P2_1, P2_2, P2_3, P2_4, P2_5, P2_6, P2_7};

 

void write_digit(byte cifra){

  byte i;

  for (i=0; i<8; i++)

  {

    digitalWrite(segment_to_pin[i], (cifra>>i)&1);

  }

}

 

void setup(){

   byte i;

   pinMode(boton_inc, INPUT_PULLUP);

   pinMode(boton_dec, INPUT_PULLUP);

   pinMode(boton_reset, INPUT_PULLUP);

   for (i=0; i<8; i++){

      pinMode(segment_to_pin[i], OUTPUT);

   }

}

 

void loop(){

   estado_boton_inc = digitalRead(boton_inc);

   estado_boton_reset = digitalRead(boton_reset);

 

   if (estado_boton_inc!=estado_boton_inc_ant)

   {

      if ((estado_boton_inc==0)&&(decenas+unidades<153))

      {

         unidades++;

      }

      if (unidades>9)

      {

         unidades = 0;

         decenas += 16;

      }

      estado_boton_inc_ant = estado_boton_inc;

      delay(5);

   }

 

   //Programar acción boton_dec

 

   if (estado_boton_reset==0)

   {

      unidades = 0;

      decenas = 0;

   }

 

   write_digit(decenas + unidades);

}
