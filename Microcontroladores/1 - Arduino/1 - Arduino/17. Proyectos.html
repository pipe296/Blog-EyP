using System;
using System.Collections.Generic;
using System.Drawing;
using System.Windows.Forms;
using System.IO.Ports;

namespace Probador_de_CI_74XX
{
    /// <summary>
    /// Description of MainForm.
    /// </summary>
    public partial class MainForm : Form
    {
        public MainForm()
        {
            //
            // The InitializeComponent() call is required for Windows Forms designer support.
            //
            InitializeComponent();
            
            //
            // TODO: Add constructor code after the InitializeComponent() call.
            //
        }

        void Button_BuscarClick(object sender, EventArgs e)
        {
            comboBox_Puertos.Items.Clear();
            foreach (string s in SerialPort.GetPortNames())
                comboBox_Puertos.Items.Add(s);
            if (SerialPort.GetPortNames().GetLength(0)>0)
                comboBox_Puertos.Enabled = true;
            serialPort1.Close();
            button1.Enabled = false;
        }

        void ComboBox_PuertosSelectedIndexChanged(object sender, EventArgs e)
        {
            if (comboBox_Puertos.Text != "")
                button_Conectar.Enabled = true;
        }

        void Button_ConectarClick(object sender, EventArgs e)
        {
            if (comboBox_Puertos.Text != "")
            {
                CheckForIllegalCrossThreadCalls = false;
                serialPort1.PortName = comboBox_Puertos.Text;
                comboBox_Puertos.Enabled = false;
                serialPort1.Open();
                button_Conectar.Enabled = false;
                button1.Enabled = true;
            }            
        }

        void Button1Click(object sender, EventArgs e)
        {
            serialPort1.Write("T" + (char)13);
        }

        void MainFormLoad(object sender, EventArgs e)
        {
            System.Threading.Thread.CurrentThread.CurrentCulture = new System.Globalization.CultureInfo("en-US");
        }

        void MainFormFormClosed(object sender, FormClosedEventArgs e)
        {
            serialPort1.Close();
            serialPort1.Dispose();
        }
        
        void SerialPort1DataReceived(object sender, SerialDataReceivedEventArgs e)
        {
            string dato;
            
            if (serialPort1.BytesToRead > 1)
            {
                dato = serialPort1.ReadLine();
                if (dato.StartsWith("AB"))
                    label1.Text = "AND Buena";
                else if (dato.StartsWith("OB"))
                    label1.Text = "OR Buena";
                else if (dato.StartsWith("DB"))
                    label1.Text = "NAND Buena";
                else if (dato.StartsWith("RB"))
                    label1.Text = "NOR Buena";
                else if (dato.StartsWith("XB"))
                    label1.Text = "XOR Buena";
                else
                    label1.Text = "Mala";
            }
        }
    }
}
