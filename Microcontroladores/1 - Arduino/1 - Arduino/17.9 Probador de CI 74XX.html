char inputString[10];
boolean stringComplete;
boolean a = 0;
boolean b = 0;
boolean c = 0;
boolean d = 0;

void setup()
{
  pinMode(8, INPUT);
  pinMode(9, OUTPUT);
  pinMode(10, OUTPUT);
  Serial.begin(9600);
}

void loop() {
   if (stringComplete)
   {
      switch (inputString[0])
      {
         case 'T': digitalWrite(9, LOW); digitalWrite(10, LOW);
                   delay(50);
                   a = digitalRead(8);
                   delay(50);
                   digitalWrite(9, LOW); digitalWrite(10, HIGH);
                   delay(50);
                   b = digitalRead(8);
                   delay(50);
                   digitalWrite(9, HIGH); digitalWrite(10, LOW);
                   delay(50);
                   c = digitalRead(8);
                   delay(50);
                   digitalWrite(9, HIGH); digitalWrite(10, HIGH);
                   delay(50);
                   d = digitalRead(8);
                   digitalWrite(9, LOW); digitalWrite(10, LOW);
                   if (a==0 && b==0 && c==0 && d==1)
                     Serial.println("AB");                  //AND Buena
                   else if (a==0 && b==1 && c==1 && d==1)
                     Serial.println("OB");                  //OR Buena
                   else if (a==1 && b==1 && c==1 && d==0)
                     Serial.println("DB");                  //NAND Buena
                   else if (a==1 && b==0 && c==0 && d==0)
                     Serial.println("RB");                  //NOR Buena
                   else if (a==0 && b==1 && c==1 && d==0)
                     Serial.println("XB");                  //XOR Buena
                   else
                     Serial.println("M");
                   break;
         default: break;
      }

      // clear the string:
      sprintf(inputString,"");
      stringComplete = false;
   }
}

void serialEvent() {
  while (Serial.available())
  {
    char inChar[2]="0";
    // get the new byte:
    inChar[0] = Serial.read();
    // add it to the inputString:
    strcat(inputString,inChar);
    // if the incoming character is a newline, set a flag
    // so the main loop can do something about it:
    if (inChar[0] == '\r')
    {
      stringComplete = true;
    }
  }
}
