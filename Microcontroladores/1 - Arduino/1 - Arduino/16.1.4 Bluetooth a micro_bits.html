// include the library code:
#include <LiquidCrystal.h>

const int rs = 12, en = 11, d4 = 5, d5 = 4, d6 = 3, d7 = 2;
LiquidCrystal lcd(rs, en, d4, d5, d6, d7);

char txt2stream[17];
char inputString[10];
boolean stringComplete;

void setup()
{
  pinMode(A0, OUTPUT);
  pinMode(A1, OUTPUT);
  pinMode(A2, OUTPUT);
  digitalWrite(10, HIGH);
  lcd.begin(16,2);
  delay(200);
  lcd.setCursor(0,0);
  lcd.print("LED:   M:   B:  ");
  Serial.begin(9600);
}

void loop() {
  if (stringComplete)
  {
    switch (inputString[0])
    {
      case '0': digitalWrite(A0, LOW); break;
      case '1': digitalWrite(A0, HIGH); break;
      default: break;
    }
    switch (inputString[1])
    {
      case '0': digitalWrite(A1, LOW); break;
      case '1': digitalWrite(A1, HIGH); break;
      default: break;
    }
      
    //Programar acción inputString[2]

    lcd.setCursor(4,0);
    sprintf(txt2stream,"%c",inputString[0]);
    lcd.print(txt2stream);

    lcd.setCursor(9,0);
    sprintf(txt2stream,"%c",inputString[1]);
    lcd.print(txt2stream);

    //Programar visualización inputString[2]

    // clear the string:
    sprintf(inputString,"");
    stringComplete = false;
  }
}

void serialEvent() {
  while (Serial.available())
  {
    char inChar[2]="0";
    // get the new byte:
    inChar[0] = Serial.read();
    // add it to the inputString:
    strcat(inputString,inChar);
    // if the incoming character is a newline, set a flag
    // so the main loop can do something about it:
    if (inChar[0] == '\r')
    {
      stringComplete = true;
    }
  }
}
