#include <Arduino.h.encrypted>

#include <LiquidCrystal.h>

 

int32 temperatura;

char grados = 223;

char txt2stream[10];

 

void setup() {

   digitalWrite(LCD_BACKLIGHT, HIGH);

   lcd_begin(16,2);

   delay(200);

   lcd_setCursor(0,0);

   lcd_print("Temp: ");

}

 

void loop() {

   temperatura = analogRead(A0);

   temperatura = temperatura * 150 / 1023;

   lcd_setCursor(6,0);

   sprintf(txt2stream,"%Lu %cC ",temperatura,grados);

   lcd_print(txt2stream);

   delay(500);

}

//#device PIC16F84A

//#device PIC16F628A

//#device PIC16F818

//#device PIC16F873A

//#device PIC16F883

//#device PIC16F877A

#device PIC16F887

//#device PIC18F2550

//#device PIC18F4455

//#device PIC18F4550

 

#include <Arduino.h.encrypted>

#include <LiquidCrystal.h>

 

#define sht_data_pin PIN_B2

#define sht_clk_pin PIN_B3

 

#include ".\Drivers\sht7x_sht1x.c"

 

float PV_Temp, PV_Hum;

char grados = 223;

char txt2stream[17];

 

void setup() {

   digitalWrite(LCD_BACKLIGHT, HIGH);

   lcd_begin(16,2);

   lcd_setCursor(0,0);

   sprintf(txt2stream,"PV_Temp:");

   lcd_print(txt2stream);

   lcd_setCursor(0,1);

   sprintf(txt2stream,"PV_Hum:");

   lcd_print(txt2stream);

   sht_init();

   Serial_begin(9600);

}

 

void loop() {

   sht_rd(PV_temp, PV_hum);

 

   lcd_setCursor(9,0);

   sprintf(txt2stream,"%.1f%cC ",PV_Temp,grados);

   lcd_print(txt2stream);

 

   lcd_setCursor(9,1);

   sprintf(txt2stream,"%.1f%% ",PV_Hum);

   lcd_print(txt2stream);

 

   sprintf(txt2stream,"$%.1f/%.1f",PV_Temp,PV_Hum);

   Serial_println(txt2stream);

   delay(50);

}

 

 

Conversor transmisor de temperatura con entrada para termocupla tipo K y salida en corriente. 

Características:

  • Rango de medición: 0°C hasta 1300°C
  • Salida de señal estándar internacional: 4~20mA
  • Precisión: 0.5±0.5% (En toda la escala)
  • Máxima salida de corriente: 26mA
  • Diámetro: 45mm
  • Material: Plástico
